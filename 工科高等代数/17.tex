\section{线性空间 IV}

\subsection{线性子空间 continue}

\begin{definition}[向量组生成的子空间]
	对于线性空间 $V$ 中的一个向量组 $\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_s$，所有由它们线性表出的向量构成一个子空间，这个子空间称为向量组 $\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_s$ 生成的子空间，记作 $L(\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_s)$。
\end{definition}

那么我们可以得到这种子空间的若干个性质。

\begin{property}
	两个向量组生成的子空间相同的充要条件是这两个向量组等价。
\end{property}

\begin{property}
	$$
	\dim L(\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_s) = \rank (\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_s)
	$$
\end{property}

\begin{property}
	子空间的任意一组基都可以通过加入原空间中的一些其它的向量扩充成原空间的基。
\end{property}

对于子空间，我们还可以定义它们的交与和：

\begin{theorem}
	对于两个子空间 $V_1, V_2$，它们的交 $V_1 \cap V_2$ 也是一个子空间。

	\begin{proof}
		因为
		$$
		\vect{\alpha} \in V_1 \land \vect{\alpha} \in V_2 \Leftrightarrow \vect{\alpha} \in V_1 \cap V_2
		$$
		而空间交中的向量在 $V_1, V_2$ 中运算均封闭，因此在 $V_1 \cap V_2$ 中运算也封闭。
	\end{proof}
\end{theorem}

可以发现，子空间的交运算满足交换律和结合律。

\begin{definition}[子空间的和]
	对于子空间 $V_1, V_2$，定义它们的和：
	$$
	V_1 + V_2 = \{\vect{\alpha}_1 + \vect{\alpha}_2: \vect{\alpha}_1 \in V_1, \vect{\alpha}_2 \in V_2\}
	$$
\end{definition}

\begin{theorem}
	对于两个子空间 $V_1, V_2$，它们的和 $V_1 + V_2$ 也是一个子空间。

	\begin{proof}
		因为 $V_1 + V_2$ 中所有向量都能表示成分别在 $V_1, V_2$ 中的一个向量的和，因此 $V_1 + V_2$ 中向量的线性表达也能表示成分别在 $V_1, V_2$ 中的线性表达的和，也就是分别在 $V_1, V_2$ 中的一个向量的和。因此有封闭性。
	\end{proof}
\end{theorem}

可以发现，子空间的和运算满足交换律和结合律。

子空间的交、和、维数可以相互联系起来，我们有定理：

\begin{theorem}[维数公式]
	对于两个子空间 $V_1, V_2$，有
	$$
	\dim V_1 + \dim V_2 = \dim (V_1 + V_2) + \dim (V_1 \cap V_2)
	$$

	\begin{proof}
		设 $\dim V_1 = n_1, \dim V_2 = n_2$。取 $V_1 \cap V_2$ 的一组基 $\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_m$，那么可以将这组基扩充使其分别称为 $V_1, V_2$ 的基：
		$$
		\begin{gathered}
			\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_m, \vect{\beta}_1, \vect{\beta}_2, \dots, \vect{\beta}_{n_1 - m} \\
			\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_m, \vect{\gamma}_1, \vect{\gamma}_2, \cdots, \vect{\gamma}_{n_2 - m}
		\end{gathered}
		$$
		那么只需要证明向量组
		$$
		\vect{\alpha}_1, \vect{\alpha}_2, \cdots, \vect{\alpha}_m, \vect{\beta}_1, \vect{\beta}_2, \dots, \vect{\beta}_{n_1 - m}, \vect{\gamma}_1, \vect{\gamma}_2, \cdots, \vect{\gamma}_{n_2 - m}
		$$
		是 $V_1 + V_2$ 的基就可以证明了。而这是很显然的。
	\end{proof}
\end{theorem}
