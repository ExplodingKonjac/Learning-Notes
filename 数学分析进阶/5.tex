\section{傅里叶级数}

\subsection{三角函数空间}

某个区间上的函数可以构成一个 Euclid 空间。其中加法和数乘就是函数的运算，而内积定义为：

\begin{definition}[函数的内积]
    对于 $[l, r]$ 上的两个函数 $f(x), g(x)$，我们定义它们的内积为
    $$
    \ab\langle f, g \rangle = \int_l^r f(x) g(x) \dd{x}
    $$
\end{definition}

有了内积之后，就可以定义正交：

\begin{definition}[正交]
    若 $\ab\langle f, g \rangle = 0$，则称 $f$ 与 $g$ 正交。
\end{definition}

三角函数系是指下面的函数线性组合得到的函数族：
$$
1,\ \sin n x,\ \cos n x,\ (n \in \mathbb{N}^+)
$$

在三角函数系中，我们定义函数的正交取区间 $[-\pi, \pi]$。那么我们可以发现：

\begin{itemize}
    \item
    $$
    \int_{-\pi}^{\pi} \sin n x \dd{x} = \int_{-\pi}^{\pi} \cos n x \dd{x} = 0
    $$
    \item
    $$
    \int_{-\pi}^{\pi} \sin m x \sin n x \dd{x} = \pi [m = n]
    $$
    \item
    $$
    \int_{-\pi}^{\pi} \cos m x \cos n x \dd{x} = \pi [m = n]
    $$
    \item
    $$
    \int_{-\pi}^{\pi} \sin m x \cos n x \dd{x} = 0
    $$
\end{itemize}

\subsection{Fourier 级数}

\begin{definition}
    一般地，\textbf{三角级数}定义为如下形式的级数：
    $$
    f(x) = \frac{a_0}{2} + \sum_{n=1}^{\infty} (a_n \cos n x + b_n \sin n x)
    $$
\end{definition}

根据函数项级数的 Weierstrass 定理，我们可以知道，若级数
$$
\sum_{n = 1}^{\infty} (\abs{a_n} + \abs{b_n})
$$
收敛，那么对应的三角级数在 $[-\pi, \pi]$ 上一致收敛且绝对收敛。

Fourier 级数要解决的问题是，对于给定的 $2\pi$ 周期函数 $f(x)$，三角级数
$$
f(x) = \frac{a_0}{2} + \sum_{n = 1}^{\infty} (a_n \cos n x + b_n \sin n x)
$$
是否存在？系数取值是什么？

我们假设存在这么一个级数一致收敛于 $f(x)$，那么：

\begin{enumerate}
    \item 求 $a_0$：
    $$
    \begin{aligned}
        \int_{-\pi}^{\pi} f(x) \dd{x} & = \int_{-\pi}^{\pi} \left( \frac{a_0}{2} + \sum_{n = 1}^{\infty} (a_n \cos n x + b_n \sin n x) \right) \dd{x} \\
        & = \int_{-\pi}^{\pi} \frac{a_0}{2} \dd{x} + \sum_{n = 1}^{\infty} \int_{-\pi}^{\pi} (a_n \cos n x + b_n \sin n x) \dd{x} \\
        & = \pi a_0
    \end{aligned}
    $$
    所以
    $$
    a_0 = \frac{1}{\pi} \int_{-\pi}^{\pi} f(x) \dd{x}
    $$
    \item 求 $a_n$：
    $$
    \begin{aligned}
        \int_{-\pi}^{\pi} f(x) \cos n x \dd{x} & = \int_{-\pi}^{\pi} \ab(\frac{a_0}{2} \cos n x + \sum_{k = 1}^{\infty} (a_k \cos k x \cos n x + b_k \sin k x \cos n x)) \dd{x} \\
        & = \frac{a_0}{2} \int_{-\pi}^{\pi} \cos n x \dd{x} + \sum_{k = 1}^{\infty} \int_{-\pi}^{\pi} (a_k \cos k x \cos n x + b_k \sin k x \cos n x) \dd{x} \\
        & = \pi a_n
    \end{aligned}
    $$
    所以
    $$
    a_n = \frac{1}{\pi} \int_{-\pi}^{\pi} f(x) \cos n x \dd{x}
    $$
    \item 求 $b_n$：
    $$
    \begin{aligned}
        \int_{-\pi}^{\pi} f(x) \sin n x \dd{x} & = \int_{-\pi}^{\pi} \ab(\frac{a_0}{2} \sin n x + \sum_{k = 1}^{\infty} (a_k \cos k x \sin n x + b_k \sin k x \sin n x)\ab) \dd{x} \\
        & = \frac{a_0}{2} \int_{-\pi}^{\pi} \sin n x \dd{x} + \sum_{k = 1}^{\infty} \int_{-\pi}^{\pi} (a_k \cos k x \sin n x + b_k \sin k x \sin n x) \dd{x} \\
        & = \pi b_n
    \end{aligned}
    $$
    所以
    $$
    b_n = \frac{1}{\pi} \int_{-\pi}^{\pi} f(x) \sin n x \dd{x}
    $$
\end{enumerate}

\begin{definition}[Fourier 级数]
    对于周期为 $2\pi$ 的周期函数 $f(x)$，定义其 \textbf{Fourier 级数}为：
    $$
    f(x) = \frac{a_0}{2} + \sum_{n = 1}^{\infty} (a_n \cos n x + b_n \sin n x)
    $$
    其中：
    $$
    \begin{gathered}
        a_n = \frac{1}{\pi} \int_{-\pi}^{\pi} f(x) \cos n x\ (n \in \mathbb{N}) \\
        b_n = \frac{1}{\pi} \int_{-\pi}^{\pi} f(x) \sin n x\ (n \in \mathbb{N}^+)
    \end{gathered}
    $$
\end{definition}

Fourier 级数的定义仅要求 $f(x)$ 可积，但是级数的取值和函数的光滑性有关。

\begin{definition}[光滑]
    称 $f(x)$ 在 $[l, r]$ \textbf{光滑}当且仅当 $f'(x)$ 在 $[l, r]$ 上连续。称 $f(x)$ 在 $[l, r]$ \textbf{分段光滑}当且仅当 $f'(x)$ 在 $[l, r]$ 仅在有限个第一类间断点处不连续。
\end{definition}

\begin{theorem}[Dirichlet 定理]
    若 $f(x)$ 是以 $2\pi$ 为周期的周期函数，且在 $[-\pi, \pi]$ 分段光滑，那么对于 $x \in [-\pi, \pi]$，$f(x)$ 的 Fourier 级数收敛于 $f(x)$ 左右极限的平均值。
\end{theorem}

观察 Fourier 级数的性质可知，当 $f(x)$ 是奇函数时，仅包含 $\sin$ 项，此时称为\textbf{正弦级数}；当 $f(x)$ 是偶函数时，仅包含 $\cos$ 项，此时称为\textbf{余弦级数}。

对于一个定义域为 $[0, \pi)$ 的函数，可以通过奇延拓或偶延拓的方式来将其展开为正弦级数或余弦级数。

\begin{problem}
    习题 13.1.2
    \begin{solution}
        \begin{enumerate}
            \item[\textbf{2)}] $f(x)$ 是偶函数，因此 Fourier 级数仅包含余弦项。
            $$
            \begin{aligned}
                a_n & = \frac{1}{\pi} \int_{-\pi}^{\pi} x \sin x \cos nx \dd{x} \\
                & = \frac{1}{\pi} \int_{-\pi}^{\pi} \frac{1}{2} x (\sin ((n + 1) x) - \sin ((n - 1) x)) \dd{x} \\
                & = \frac{1}{2 \pi} \ab(\int_{-\pi}^{\pi} x \sin ((n + 1) x) \dd{x} - \int_{-\pi}^{\pi} x \sin ((n - 1) x) \dd{x})
            \end{aligned}
            $$
            对于 $k \neq 0$ 有：
            $$
            \begin{aligned}
                \int_{-\pi}^{\pi} x \sin k x \dd{x} & = -\int_{-\pi}^{\pi} \frac{x}{k} \dd(\cos k x) \\
                & = -\ab[\frac{x}{k} \cos k x]_{-\pi}^{\pi} + \int_{-\pi}^{\pi} \frac{1}{k} \cos k x \dd{x} \\
                & = -\frac{2\pi}{k} \cos k \pi + \ab[\frac{1}{k^2} \sin k x]_{-\pi}^{\pi} \\
                & = -\frac{2\pi}{k} \cos k \pi = -\frac{2\pi (-1)^{k}}{k}
            \end{aligned}
            $$
            因此：
            $$
            a_n = \begin{dcases}
                \frac{2 (-1)^{n + 1}}{n^2 - 1} & , n \neq 1 \\
                -\frac{1}{2} & , n = 1
            \end{dcases}
            $$
            所以 $f(x)$ 的 Fourier 级数是
            $$
            f(x) = 1 - \frac{1}{2} \cos x + \sum_{n = 2}^{\infty} \frac{2 (-1)^{n + 1}}{n^2 - 1} \cos n x
            $$

            \item[\textbf{3)}] 有：
            $$
            \begin{aligned}
                \int x^2 \cos n x \dd{x} & = \frac{x^2}{n} \sin n x + \frac{2 x}{n^2} \cos n x - \frac{2}{n^3} \sin n x + C \\
                \int x^2 \sin n x \dd{x} & = -\frac{x^2}{n} \cos n x + \frac{2 x}{n^2} \sin n x + \frac{2}{n^3} \cos n x + C
            \end{aligned}
            $$
            因此：
            $$
            \begin{aligned}
                a_0 & = \frac{1}{\pi} \int_{0}^{2\pi} f(x) \cos 0x \dd{x} = \frac{1}{\pi} \ab(\int_{0}^{\pi} x^2 \dd{x} - \int_{\pi}^{2\pi} x^2 \dd{x}) = -2\pi^2 \\
                a_n & = \frac{1}{\pi} \int_{0}^{2\pi} f(x) \cos n x \dd{x} \\
                & = \frac{1}{\pi} \ab(\int_{0}^{\pi} x^2 \cos n x \dd{x} - \int_{\pi}^{2\pi} x^2 \cos n x \dd{x}) \\
                & = \frac{4 ((-1)^{n} - 1)}{n^2} \\
                b_n & = \frac{1}{\pi} \int_{0}^{2\pi} f(x) \sin n x \dd{x} \\
                & = \frac{1}{\pi} \ab(\int_{0}^{\pi} x^2 \sin n x \dd{x} - \int_{\pi}^{2\pi} x^2 \sin n x \dd{x}) \\
                & = \frac{4 ((-1)^{n} - 1)}{\pi n^3} - \frac{2\pi ((-1)^{n} - 2)}{n}
            \end{aligned}
            $$
            所以 $f(x)$ 的 Fourier 级数为
            $$
            f(x) = -\pi^2 + \sum_{n = 1}^{\infty} \ab(\frac{4 ((-1)^{n} - 1)}{n^2} \cos n x + \ab(\frac{4 ((-1)^{n} - 1)}{\pi n^3} - \frac{2\pi ((-1)^{n} - 2)}{n}) \sin n x)
            $$

            \item[\textbf{4)}] $f(x)$ 是一个偶函数，因此仅包含余弦项。
            $$
            \begin{aligned}
                a_0 & = \frac{1}{\pi} \int_{-\pi}^{\pi} \abs{x} \dd{x} = \frac{2}{\pi} \int_0^{\pi} x \dd{x} = \pi \\
                a_n & = \frac{1}{\pi} \int_{-\pi}^{\pi} \abs{x} \cos n x \dd{x} = \frac{2}{\pi} \int_0^{\pi} x \cos n x \dd{x} \\
                & = \frac{2 ((-1)^{n} - 1)}{\pi n^2}
            \end{aligned}
            $$
            因此 $f(x)$ 的 Fourier 级数为
            $$
            f(x) = \frac{\pi}{2} + \sum_{n = 1}^{\infty} \frac{2 ((-1)^{n} - 1)}{\pi n^2} \cos n x
            $$
        \end{enumerate}
    \end{solution}
\end{problem}

\begin{problem}
    习题 13.1.3
    \begin{solution}
        \begin{enumerate}
            \item[\textbf{1)}]
            $$
            \begin{aligned}
                \int_{-\pi}^{0} f(x) \sin (2n - 1)x \dd{x} & = \int_{0}^{\pi} f(x - \pi) \sin (2n - 1) (x - \pi) \dd{x} \\
                & = \int_{0}^{\pi} f(x) \sin ((2n - 1) x - (2n - 1)\pi) \dd{x} \\
                & = -\int_{0}^{\pi} f(x) \sin (2n - 1) x \dd{x} \\
                \implies \int_{-\pi}^{\pi} f(x) \sin (2n - 1) \dd{x} & = 0
            \end{aligned}
            $$
            同理可得
            $$
            \int_{-\pi}^{\pi} f(x) \cos (2n - 1) x \dd{x} = 0
            $$
            因此 $a_{2n - 1} = b_{2n - 1} = 0$。

            \item[\textbf{2)}]
            $$
            \begin{aligned}
                \int_{-\pi}^{0} f(x) \sin 2n x \dd{x} & = \int_{0}^{\pi} f(x - \pi) \sin 2n (x - \pi) \dd{x} \\
                & = \int_{0}^{\pi} -f(x) \sin (2n x - 2n \pi) \dd{x} \\
                & = -\int_{0}^{\pi} f(x) \sin 2n x \dd{x} \\
                \implies \int_{-\pi}^{\pi} f(x) \sin 2n x \dd{x} & = 0
            \end{aligned}
            $$
            同理可得
            $$
            \int_{-\pi}^{\pi} f(x) \cos 2n x \dd{x} = 0
            $$
            因此 $a_{2n} = b_{2n} = 0$。
        \end{enumerate}
    \end{solution}
\end{problem}

\begin{problem}
    习题 13.1.4
    \begin{solution}
        \begin{enumerate}
            \item[\textbf{1)}] 将函数沿拓为：
            $$
            \hat{f}(x) = \frac{\pi}{2} - \abs{x}
            $$
            那么：
            $$
            \begin{aligned}
                a_0 & = \frac{1}{\pi} \int_{-\pi}^{\pi} \hat{f}(x) \dd{x} = 0 \\
                a_n & = \frac{1}{\pi} \int_{-\pi}^{\pi} \hat{f}(x) \cos n x \dd{x} \\
                & = \frac{2}{\pi} \int_{0}^{\pi} \ab(\frac{\pi}{2} - x) \cos n x \dd{x} \\
                & = \frac{2 (1 - (-1)^{n})}{\pi n^2}
            \end{aligned}
            $$
            因此 $f(x)$ 的 Fourier 级数为
            $$
            f(x) = \sum_{n = 1}^{\infty} \frac{2 (1 - (-1)^{n})}{\pi n^2} \cos n x
            $$

            \item[\textbf{2)}] 将函数沿拓为：
            $$
            \hat{f}(x) = \sgn x \cos \frac{x}{2}
            $$
            那么：
            $$
            \begin{aligned}
                b_n & = \frac{1}{\pi} \int_{-\pi}^{\pi} \hat{f}(x) \dd{x} \\
                & = \frac{2}{\pi} \int_{0}^{\pi} \cos \frac{x}{2} \sin n x \dd{x} \\
                & = \frac{8 n}{(4n^2 - 1)\pi}
            \end{aligned}
            $$
            因此 $f(x)$ 的 Fourier 级数为
            $$
            f(x) = \sum_{n = 1}^{\infty} \frac{8 n}{(4n^2 - 1)\pi} \sin n x
            $$

            \item[\textbf{4)}]
            $$
            \begin{aligned}
                a_0 & = \frac{2}{\pi} \int_{0}^{\pi} (1 + x) \dd{x} = 2 + \pi \\
                a_n & = \frac{2}{\pi} \int_{0}^{\pi} (1 + x) \cos n x \dd{x} = \frac{2 ((-1)^{n} - 1)}{\pi n^2} \\
                b_n & = \frac{2 (1 - (1 + \pi)(-1)^{n})}{\pi n}
            \end{aligned}
            $$
            因此 $f(x)$ 的正弦级数为
            $$
            f(x) = \sum_{n = 1}^{\infty} \frac{2(1 - (1 + \pi)(-1)^{n})}{\pi n} \sin n x
            $$
            余弦级数为
            $$
            f(x) = \frac{2 + \pi}{2} + \sum_{n = 1}^{\infty} \frac{2((-1)^{n} - 1)}{\pi n^2} \cos n x
            $$
        \end{enumerate}
    \end{solution}
\end{problem}

            $$

            $$
            代入 $x = 0$ 得到：

            \sum_{n = 1}^{\infty} \frac{1}{4n^2 - 1} = \frac{1}{2}
            $$

                a_0 & = \frac{1}{12\pi} \int_{0}^{2\pi} (2x^2 - 6\pi x + 2\pi^2) \dd{x} = 0 \\

\begin{problem}
    习题 13.1.7
    \begin{solution}
        \begin{enumerate}
            \item[\textbf{1)}] $f(x)$ 是一个偶函数，仅包含 $\cos$ 项。
            $$
            \begin{aligned}
                a_0 & = \frac{1}{\pi} \int_{-\pi}^{\pi} \cos \alpha x \dd{x} = \frac{2 \sin \alpha \pi}{\alpha \pi} \\
                a_n & = \frac{1}{\pi} \ab(\frac{\sin (\alpha \pi + n \pi)}{\alpha + n} + \frac{\sin (\alpha \pi - n \pi)}{\alpha - n}) \\
                & = \frac{2 \alpha (-1)^{n} \sin \alpha \pi}{\pi(\alpha^2 - n^2)}
            \end{aligned}
            $$
            因此 $f(x)$ 的 Fourier 级数为
            $$
            f(x) = \frac{\sin \alpha \pi}{\alpha \pi} + \sum_{n = 1}^{\infty} \frac{2 \alpha (-1)^{n} \sin \alpha \pi}{\pi(\alpha^2 - n^2)} \cos n x
            $$

            \item[\textbf{2)}] 化简 $f(x)$ 的 Fourier 级数：
            $$
            f(x) = \frac{\sin \alpha \pi}{\pi} \ab(\frac{1}{\alpha} + \sum_{n = 1}^{\infty} \frac{2 \alpha (-1)^{n}}{\alpha^2 - n^2} \cos n x)
            $$
            代入 $x = 0, \alpha = a$ 即得
            $$
            f(0) = \frac{\sin a \pi}{\pi} \ab(\frac{1}{a} + \sum_{n = 1}^{\infty} \frac{2 a}{a^2 - n^2}) = 1 \implies \frac{1}{a} + \sum_{n = 1}^{\infty} \frac{2 a}{a^2 - n^2} = \frac{\pi}{\sin a \pi}
            $$

            \item[\textbf{3)}] 令 $x \gets 0, \alpha \gets \frac{x}{\pi}$ 代入 $f(x)$ 的 Fourier 级数：
            $$
            \frac{\sin x}{\pi} \ab(\frac{\pi}{x} + \sum_{n = 1}^{\infty} \frac{2 x \pi}{x^2 - n^2 \pi^2} (-1)^{n}) = f(0) = 1 \implies \frac{1}{x} + \sum_{n = 1}^{\infty} \frac{2 x}{x^2 - n^2 \pi^2} (-1)^{n} = \frac{1}{\sin x}
            $$
        \end{enumerate}
    \end{solution}
\end{problem}
