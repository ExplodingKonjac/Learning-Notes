\section{一阶谓词逻辑 II}

\subsection{前束范式}

\begin{definition}[前束范式]
	设 $\delta_k$ 为 $\forall, \exists$ 量词，$x_1, x_2, \dots, x_n$ 为不同变元，$Q$ 为开公式，那么形式为
	$$
	\delta_1\, x_1\, \delta_2\, x_2\, \dots\, \delta_n\, x_n\, Q
	$$
	的公式称为\textbf{前束范式}，成 $\delta_1\, x_1\, \dots\, \delta_n\, x_n$ 为前束词，$Q$ 为母式。
\end{definition}
